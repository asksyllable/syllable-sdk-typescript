# Event

An event represents a specific occurrence during a session.

## Example Usage

```typescript
import { Event } from "syllable-sdk/models/components";

let value: Event = {
  timestamp: new Date("2025-09-20T14:02:09.066Z"),
  source: "tool-call",
  category: "automation",
  type: "phone-transfer",
  userId: "+18042221111",
  description: "Transfer to phone number +18042221111",
};
```
