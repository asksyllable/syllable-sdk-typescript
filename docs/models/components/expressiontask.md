# ExpressionTask

## Example Usage

```typescript
import { ExpressionTask } from "syllable-sdk/models/components";

let value: ExpressionTask = {
  variables: [
    {
      valueFrom: "inputs.provided_dob == patient_dob",
      name: "<value>",
    },
  ],
  type: "expression",
  expression: "inputs.can_sign_consent == `true`",
  on: {
    start: [
      {
        action: "set",
        name: "<value>",
      },
    ],
    submit: [
      {
        action: "inc",
        name: "<value>",
        by: 1,
      },
    ],
  },
};
```
