# TimelineEvent

A single normalized event on the consolidated session timeline.

## Example Usage

```typescript
import { TimelineEvent } from "syllable-sdk/models/components";

let value: TimelineEvent = {
  kind: "tool",
  timestamp: new Date("2026-08-12T01:40:01.547Z"),
  offset: "00:04",
  category: "tool",
  label: "holiday_check",
  metadata: [
    "Gpt-4.1",
    "OpenAI",
  ],
};
```
