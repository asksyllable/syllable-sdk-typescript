# InsightsOutput

Response model for an insight tool.

## Example Usage

```typescript
import { InsightsOutput } from "syllable-sdk/models/components";

let value: InsightsOutput = {
  id: 1,
  sessionId: 12341,
  uploadFileId: 12535,
  insightToolId: 34531,
  insightToolVersion: 1,
  insightKey: "summary",
  stringValue: "This is a summary of the call",
  numericValue: 1,
  jsonValue: {
    "content":
      " was there anything else that I can assist you with other than this",
    "role": "assistant",
    "start_time": 289,
  },
  createdAt: new Date("2025-07-22T00:00:00Z"),
  updatedAt: new Date("2025-07-23T00:00:00Z"),
};
```
